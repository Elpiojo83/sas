### Includes - setup ###
### HTML5 Readykit ###

# Include CSS files
page.includeCSS{
	
	# This is the default HTML5 Boilerplate stylesheet. This should be the first stylesheet in the header
	# There are more css files in the Boilerplate css folder (EXT:html5boilerplate/res/css)
	boilerplate = EXT:html5_readykit/res/css/style.css
	
	# Your main css file can be inserted here
	main = {$plugin.html5_readykit.includeAdditionalCSS}
		
}


# Include Modernizr as the last script in the header
page.headerData.950 = TEXT
page.headerData.950.value = <script src="typo3conf/ext/html5_readykit/res/js/libs/modernizr-2.0.min.js"></script>

# Include jQuery from Google Libraries API
page.includeJS{
	jquery = //ajax.googleapis.com/ajax/libs/jquery/1.6.2/jquery.min.js
	jquery.external = 1		
}

# Include main js file
page.includeJS{
	main = {$plugin.html5_readykit.includeAdditionalJS}
}

# Insert a PNG fix for IE6 and less
[browser = msie] && [version =< 7]
page.footerData.900 = TEXT
page.footerData.900.value(
		<script src="typo3conf/ext/html5_readykit/res/js/libs/dd_belatedpng.js"></script>
		<script> DD_belatedPNG.fix('img, .png_bg'); //fix any <img> or .png_bg background-images </script>
)
[global]
